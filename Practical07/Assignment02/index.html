Step 1

Compute Validation Metrics

Python
import numpy as np
import tensorflow as tf
from sklearn.metrics import (
    accuracy_score, precision_score,
    recall_score, f1_score, classification_report
)

CLASS_NAMES = ['apple', 'banana', 'mango']

# Load saved model
model = tf.keras.models.load_model('mobilenet_3class_final.keras')

# Collect all predictions from validation generator
all_true, all_pred = [], []
val_gen.reset()

for images, labels in val_gen:
    preds = model.predict(images, verbose=0)
    all_pred.extend(np.argmax(preds, axis=1))
    all_true.extend(np.argmax(labels, axis=1))
    if len(all_true) >= val_gen.n:
        break

all_true = np.array(all_true)
all_pred = np.array(all_pred)

# Core metrics
acc  = accuracy_score(all_true, all_pred)
prec = precision_score(all_true, all_pred, average='weighted', zero_division=0)
rec  = recall_score(all_true, all_pred, average='weighted')
f1   = f1_score(all_true, all_pred, average='weighted')

print(f"Accuracy  : {acc*100:.2f}%")
print(f"Precision : {prec*100:.2f}%")
print(f"Recall    : {rec*100:.2f}%")
print(f"F1 Score  : {f1*100:.2f}%")

# Full per-class breakdown
print("\n" + "="*55)
print(classification_report(all_true, all_pred, target_names=CLASS_NAMES, digits=4))
Step 2

Build & Plot Confusion Matrix

Python
from sklearn.metrics import confusion_matrix
import seaborn as sns
import matplotlib.pyplot as plt
import matplotlib.gridspec as gridspec

cm      = confusion_matrix(all_true, all_pred)
cm_norm = cm.astype(float) / cm.sum(axis=1, keepdims=True)

fig = plt.figure(figsize=(18, 7), facecolor='#0d0618')
gs  = gridspec.GridSpec(1, 2, figure=fig, wspace=.4)

for idx, (data, title, fmt, vmax) in enumerate([
    (cm,       'Absolute Counts',      'd',  None),
    (cm_norm,  'Normalized (row %)',  '.1%', 1.0)
]):
    ax = fig.add_subplot(gs[idx])
    ax.set_facecolor('#150924')

    sns.heatmap(
        data,
        annot=True, fmt=fmt,
        xticklabels=CLASS_NAMES,
        yticklabels=CLASS_NAMES,
        cmap=sns.color_palette("magma", as_cmap=True),
        vmin=0, vmax=vmax,
        linewidths=.8,
        linecolor='#2a1040',
        annot_kws={"size": 15, "weight": "bold", "color": "white"},
        cbar_kws={"shrink": .75, "pad": .02},
        ax=ax
    )
    ax.set_title(title, color='white', fontsize=14, pad=14, fontweight='bold')
    ax.set_xlabel('Predicted Label', color='#b39ddb', labelpad=10)
    ax.set_ylabel('True Label',      color='#b39ddb', labelpad=10)
    ax.tick_params(colors='#9e9e9e', labelsize=11)
    for spine in ax.spines.values():
        spine.set_edgecolor('#3a1060')

plt.suptitle('Confusion Matrix — 3-Class Fruit Classifier',
             color='white', fontsize=16, y=1.02, fontweight='bold')
plt.savefig('confusion_matrix.png', dpi=150, bbox_inches='tight',
            facecolor=fig.get_facecolor())
plt.show()
print("Saved → confusion_matrix.png")
Step 3

ROC Curves & AUC Scores

Python
from sklearn.metrics import roc_curve, auc
from sklearn.preprocessing import label_binarize

# Binarize for one-vs-rest ROC
y_bin = label_binarize(all_true, classes=[0, 1, 2])

# Collect raw softmax probabilities (not argmax)
all_probs = []
val_gen.reset()
for images, _ in val_gen:
    probs = model.predict(images, verbose=0)
    all_probs.extend(probs)
    if len(all_probs) >= val_gen.n:
        break
all_probs = np.array(all_probs[:val_gen.n])

# Plot
palette = ['#ea80fc', '#82b1ff', '#ccff90']
fig, ax = plt.subplots(figsize=(8, 7), facecolor='#0d0618')
ax.set_facecolor('#150924')

for i, (cls, col) in enumerate(zip(CLASS_NAMES, palette)):
    fpr, tpr, _ = roc_curve(y_bin[:, i], all_probs[:, i])
    roc_auc     = auc(fpr, tpr)
    ax.plot(fpr, tpr, color=col, lw=2.5,
            label=f'{cls.title()} (AUC = {roc_auc:.3f})')

ax.plot([0,1], [0,1], 'w--', lw=1, alpha=.4, label='Random (AUC=0.500)')
ax.set_xlim([0,1]); ax.set_ylim([0,1.02])
ax.set_xlabel('False Positive Rate', color='#b39ddb')
ax.set_ylabel('True Positive Rate',  color='#b39ddb')
ax.set_title('ROC Curves per Class', color='white', fontsize=14, pad=12)
ax.legend(facecolor='#1e0930', edgecolor='#5e2a90',
          labelcolor='white', fontsize=11)
ax.tick_params(colors='#888')
for sp in ax.spines.values():
    sp.set_color('#3a1060')
plt.tight_layout()
plt.savefig('roc_curves.png', dpi=150, bbox_inches='tight',
            facecolor=fig.get_facecolor())
plt.show()

Analysis

How to Read a Confusion Matrix

TP

True Positives — Diagonal

Cells on the main diagonal are correct predictions. A perfect model has all mass on the diagonal and zeros everywhere else.

FP

False Positives — Columns

Off-diagonal values in a column show cases where that class was incorrectly predicted. High column values = over-prediction.

FN

False Negatives — Rows

Off-diagonal values in a row show true instances that were missed. High row values = model is biased away from this class.

F1

F1 Score — Harmonic Mean

F1 = 2·(P·R)/(P+R). Key metric for imbalanced datasets — accuracy alone can be misleading when one class dominates the data.
