import tensorflow as tf
from tensorflow.keras.preprocessing.image import ImageDataGenerator

IMG_SIZE    = 224
BATCH_SIZE  = 32
NUM_CLASSES = 3   # apple, banana, mango
DATA_DIR    = "fruits/"

datagen = ImageDataGenerator(
    rescale=1./255,
    rotation_range=25,
    width_shift_range=0.15,
    height_shift_range=0.15,
    shear_range=0.1,
    zoom_range=0.2,
    horizontal_flip=True,
    brightness_range=[0.8, 1.2],
    fill_mode='nearest',
    validation_split=0.2
)

train_gen = datagen.flow_from_directory(
    DATA_DIR,
    target_size=(IMG_SIZE, IMG_SIZE),
    batch_size=BATCH_SIZE,
    class_mode='categorical',
    subset='training',
    shuffle=True
)
val_gen = datagen.flow_from_directory(
    DATA_DIR,
    target_size=(IMG_SIZE, IMG_SIZE),
    batch_size=BATCH_SIZE,
    class_mode='categorical',
    subset='validation',
    shuffle=False
)

print("Classes:", train_gen.class_indices)
# → {'apple': 0, 'banana': 1, 'mango': 2}

from tensorflow.keras.applications import MobileNetV2
from tensorflow.keras import layers, models
import tensorflow as tf

def build_transfer_model(num_classes=3):
    # Load pretrained base — NO top classifier
    base = MobileNetV2(
        input_shape=(224, 224, 3),
        include_top=False,
        weights='imagenet'
    )
    base.trainable = False   # freeze all 154 layers

    # Custom classification head
    inputs = tf.keras.Input(shape=(224, 224, 3))
    x = base(inputs, training=False)
    x = layers.GlobalAveragePooling2D()(x)
    x = layers.BatchNormalization()(x)
    x = layers.Dense(256, activation='relu',
                     kernel_regularizer=tf.keras.regularizers.l2(1e-4))(x)
    x = layers.Dropout(0.4)(x)
    x = layers.Dense(64, activation='relu')(x)
    x = layers.Dropout(0.2)(x)
    outputs = layers.Dense(num_classes, activation='softmax')(x)

    model = models.Model(inputs, outputs)

    trainable = sum(tf.size(w).numpy() for w in model.trainable_variables)
    total     = sum(tf.size(w).numpy() for w in model.variables)
    print(f"Trainable: {trainable:,} / Total: {total:,}")
    return model

model = build_transfer_model(NUM_CLASSES)

# Phase 2 — fine-tune top layers of MobileNetV2 base
base_model = model.layers[1]   # the MobileNetV2 layer
base_model.trainable = True

# Freeze all layers except the last 30
for layer in base_model.layers[:-30]:
    layer.trainable = False

# Check how many are now trainable
trainable_count = sum(1 for l in base_model.layers if l.trainable)
print(f"Fine-tuning {trainable_count} base layers")

# Recompile with a very small learning rate
model.compile(
    optimizer=tf.keras.optimizers.Adam(learning_rate=1e-5),
    loss='categorical_crossentropy',
    metrics=['accuracy']
)

callbacks_ft = [
    tf.keras.callbacks.EarlyStopping(
        patience=5, restore_best_weights=True
    ),

import numpy as np
from tensorflow.keras.preprocessing import image
import tensorflow as tf

CLASS_NAMES = ['apple', 'banana', 'mango']

# Load the saved model
model = tf.keras.models.load_model('mobilenet_3class_final.keras')

def predict_fruit(img_path, top_k=3):
    """Predict fruit class with top-K confidence scores."""
    img = image.load_img(img_path, target_size=(224, 224))
    arr = image.img_to_array(img) / 255.0
    arr = np.expand_dims(arr, axis=0)

    probs   = model.predict(arr, verbose=0)[0]
    top_idx = np.argsort(probs)[::-1][:top_k]

    results = []
    for idx in top_idx:
        results.append({
            'class':      CLASS_NAMES[idx],
            'confidence': float(probs[idx]),
            'percentage': f"{probs[idx]*100:.1f}%"
        })
    return results

# Example
results = predict_fruit("test_images/banana.jpg")
for r in results:
    print(f"{r['class']:10s} → {r['percentage']}")

# banana     → 97.3%
# apple      → 1.9%
# mango      → 0.8%

Theory

Key Concepts

🧠

Feature Reuse

MobileNet's frozen layers detect edges → textures → shapes — features that transfer perfectly to fruit recognition without relearning.

🔒

Layer Freezing

Prevents catastrophic forgetting. Frozen weights stay fixed; only the custom head trains in Phase 1, preserving all 3.4M ImageNet representations.

📐

224×224 Input

MobileNetV2 requires exactly 224×224×3 RGB input, normalized to [0,1]. All images are resized automatically by ImageDataGenerator.

⚖️

Class Balance

Equal class sizes prevent bias. For imbalanced datasets, use class_weight parameter in model.fit() to compensate during loss computation.

💧

Dropout

Two dropout layers (0.4 and 0.2) randomly zero activations during training — forces redundant representations, drastically reduces overfitting.

📉

Two-Phase LR

Phase 1 uses Adam(1e-3) for fast head convergence. Phase 2 drops to Adam(1e-5) to gently nudge base layers without overwriting ImageNet knowledge.
