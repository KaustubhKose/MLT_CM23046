import tensorflow as tf
from tensorflow.keras.applications import MobileNetV2
from tensorflow.keras import layers, models

CLASS_NAMES_V1 = ['apple', 'banana', 'mango']

def make_model(num_classes):
    """Reusable model builder — swap num_classes for each phase."""
    base = MobileNetV2(
        input_shape=(224, 224, 3),
        include_top=False,
        weights='imagenet'
    )
    base.trainable = False
    x       = layers.GlobalAveragePooling2D()(base.output)
    x       = layers.BatchNormalization()(x)
    x       = layers.Dense(256, activation='relu')(x)
    x       = layers.Dropout(0.35)(x)
    outputs = layers.Dense(num_classes, activation='softmax')(x)
    return models.Model(base.input, outputs), base

📊 Compare all phases & plot performance trajectory
import matplotlib.pyplot as plt
import numpy as np
from sklearn.metrics import classification_report
import tensorflow as tf

models_cfg = {
    '3-Class': ('fruit_v1_3class.keras', val_gen_3, CLASS_NAMES_V1),
    '5-Class': ('fruit_v2_5class.keras', val_gen_5, CLASS_NAMES_V2),
    '7-Class': ('fruit_v3_7class.keras', val_gen_7, CLASS_NAMES_V3),
}

results = {}
for label, (path, vgen, cnames) in models_cfg.items():
    m       = tf.keras.models.load_model(path)
    y_true, y_pred = [], []
    vgen.reset()
    for imgs, lbls in vgen:
        y_pred.extend(np.argmax(m.predict(imgs, verbose=0), axis=1))
        y_true.extend(np.argmax(lbls, axis=1))
        if len(y_true) >= vgen.n: break
    rep = classification_report(y_true[:vgen.n], y_pred[:vgen.n],
                                 target_names=cnames, output_dict=True)
    results[label] = rep

# Plot accuracy trajectory + per-class F1 for final model
n_classes = [3, 5, 7]
accs  = [results[k]['accuracy'] * 100 for k in results]
macro = [results[k]['macro avg']['f1-score'] * 100 for k in results]

fig, axes = plt.subplots(1, 2, figsize=(15, 6), facecolor='#0c0804')
for ax in axes: ax.set_facecolor('#160e06')

axes[0].plot(n_classes, accs,  'o-', color='#f57c00', lw=2.5, ms=10, mfc='#ffb74d', label='Val Accuracy')
axes[0].plot(n_classes, macro, 's--', color='#ff8a65', lw=2, ms=8, mfc='#ffccbc', label='Macro F1')
axes[0].set_xlabel('Number of Classes', color='#ffb74d')
axes[0].set_ylabel('Score (%)', color='#ffb74d')
axes[0].set_title('Performance vs Class Count', color='white', fontsize=13)
axes[0].legend(facecolor='#1e1006', edgecolor='#5c3200', labelcolor='white')
axes[0].grid(color='rgba(245,124,0,.1)', linestyle='--')
axes[0].tick_params(colors='#777')

f1_7 = [results['7-Class'][c]['f1-score']*100 for c in CLASS_NAMES_V3]
bars = axes[1].barh(CLASS_NAMES_V3, f1_7, color='#f57c00', alpha=.85, height=.6)
axes[1].set_xlabel('F1 Score (%)', color='#ffb74d')
axes[1].set_title('7-Class: Per-Class F1', color='white', fontsize=13)
axes[1].set_xlim([60, 100])
axes[1].tick_params(colors='#888')
for bar, val in zip(bars, f1_7):
    axes[1].text(bar.get_width() + .5, bar.get_y() + bar.get_height()/2,
                 f'{val:.1f}%', va='center', color='#ffb74d', fontsize=10)

plt.suptitle('Incremental Retraining Analysis', color='white', fontsize=15)
plt.tight_layout()
plt.savefig('performance_comparison.png', dpi=150, bbox_inches='tight',
            facecolor=fig.get_facecolor())
plt.show()

FINDINGS

Key Observations

📉

Accuracy Drops Incrementally

Each expansion increases inter-class confusion. 3→5 drops ~2.5%; 5→7 drops ~2.3%. Expected and manageable with sufficient per-class data.

🔄

Weight Transfer Prevents Forgetting

Reusing the MobileNet base preserves ImageNet visual features. Training from scratch each phase would require 3× more data and compute to match.

🎯

Similar Classes Confuse More

Peach and mango (orange-toned), grape and blueberry share textures. Solution: collect diverse lighting/angle images for visually similar classes.

⚙️

Fine-tune Depth Should Scale

Phase 1 fine-tunes 20 base layers; Phase 3 fine-tunes 30. As class diversity grows, deeper adaptation is needed — monitor val loss to prevent overfitting.
