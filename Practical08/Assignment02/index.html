Step 1

Compute Validation Accuracy

Python 
import numpy as np
import tensorflow as tf
from sklearn.metrics import (
    accuracy_score, precision_score,
    recall_score, f1_score, classification_report
)

CLASS_NAMES = ['apple', 'banana', 'mango']

# Load saved model
model = tf.keras.models.load_model('mobilenet_3class.keras')

# Collect true labels and predictions from validation set
all_true, all_pred = [], []

for images, labels in val_ds:
    preds = model.predict(images, verbose=0)
    all_pred.extend(np.argmax(preds, axis=1))
    all_true.extend(np.argmax(labels.numpy(), axis=1))

all_true = np.array(all_true)
all_pred = np.array(all_pred)

# Core metrics
acc   = accuracy_score(all_true, all_pred)
prec  = precision_score(all_true, all_pred, average='weighted')
rec   = recall_score(all_true, all_pred, average='weighted')
f1    = f1_score(all_true, all_pred, average='weighted')

print(f"Accuracy  : {acc*100:.2f}%")
print(f"Precision : {prec*100:.2f}%")
print(f"Recall    : {rec*100:.2f}%")
print(f"F1 Score  : {f1*100:.2f}%")

# Detailed per-class report
print("\n" + classification_report(all_true, all_pred, target_names=CLASS_NAMES))
Step 2

Build Confusion Matrix

Python 
from sklearn.metrics import confusion_matrix
import seaborn as sns
import matplotlib.pyplot as plt
import matplotlib.ticker as ticker

cm = confusion_matrix(all_true, all_pred)

# Normalize per row (percentage of true class)
cm_norm = cm.astype(float) / cm.sum(axis=1, keepdims=True)

fig, axes = plt.subplots(1, 2, figsize=(16, 6))
fig.patch.set_facecolor('#1a0a10')

for ax, data, title, fmt in zip(
    axes,
    [cm,       cm_norm],
    ['Counts', 'Normalized (%)'],
    ['d',      '.1%']
):
    sns.heatmap(
        data,
        annot=True, fmt=fmt,
        xticklabels=CLASS_NAMES,
        yticklabels=CLASS_NAMES,
        cmap='RdPu',
        linewidths=.5,
        linecolor='#2a0d18',
        cbar_kws={'shrink': .8},
        ax=ax
    )
    ax.set_title(title, color='white', fontsize=14, pad=12)
    ax.set_xlabel('Predicted Label', color='#ccc')
    ax.set_ylabel('True Label', color='#ccc')
    ax.tick_params(colors='#aaa')

plt.suptitle('Confusion Matrix — 3-Class Fruit Classifier',
             color='white', fontsize=16, y=1.02)
plt.tight_layout()
plt.savefig('confusion_matrix.png', dpi=150,
            bbox_inches='tight', facecolor=fig.get_facecolor())
plt.show()
Step 3

ROC Curves & AUC

Python 
from sklearn.metrics import roc_curve, auc
from sklearn.preprocessing import label_binarize
import matplotlib.pyplot as plt

# Binarize labels for one-vs-rest ROC
y_true_bin = label_binarize(all_true, classes=[0, 1, 2])

# Collect raw probabilities (not argmax)
all_prob = []
for images, _ in val_ds:
    prob = model.predict(images, verbose=0)
    all_prob.extend(prob)
all_prob = np.array(all_prob)

colors = ['#f48fb1', '#ce93d8', '#ffab91']

fig, ax = plt.subplots(figsize=(8, 7))
fig.patch.set_facecolor('#1a0a10')
ax.set_facecolor('#1a0a10')

for i, (cls, col) in enumerate(zip(CLASS_NAMES, colors)):
    fpr, tpr, _ = roc_curve(y_true_bin[:, i], all_prob[:, i])
    roc_auc = auc(fpr, tpr)
    ax.plot(fpr, tpr, color=col, lw=2.5,
            label=f'{cls.title()} (AUC = {roc_auc:.3f})')

ax.plot([0, 1], [0, 1], 'w--', lw=1, alpha=.4)
ax.set_xlim([0, 1]); ax.set_ylim([0, 1.02])
ax.set_xlabel('False Positive Rate', color='#ccc')
ax.set_ylabel('True Positive Rate', color='#ccc')
ax.set_title('ROC Curves — Per Class', color='white', fontsize=14)
ax.legend(loc='lower right', facecolor='#2a0d18', edgecolor='#5a1a35', labelcolor='white')
ax.tick_params(colors='#888')
ax.spines[:].set_color('#3a1020')
plt.tight_layout()
plt.savefig('roc_curves.png', dpi=150, bbox_inches='tight', facecolor=fig.get_facecolor())
plt.show()

Analysis

How to Read Results

TP

True Positives (Diagonal)

Cells on the main diagonal are correct predictions. A perfect classifier would have all mass on the diagonal and zeros everywhere else.

FP

False Positives (Columns)

Off-diagonal values in a column show how often that class was incorrectly predicted. High column values indicate a class is being over-predicted.

FN

False Negatives (Rows)

Off-diagonal values in a row show true instances that were missed. High row values indicate a class is under-detected — the model is biased away from it.

F1

F1 Score

Harmonic mean of precision and recall. F1 is the key metric when class sizes are imbalanced — accuracy alone can be misleading when one class dominates.
