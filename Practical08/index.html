Python 
import tensorflow as tf
from tensorflow.keras.preprocessing.image import ImageDataGenerator

IMG_SIZE    = 224
BATCH_SIZE  = 32
NUM_CLASSES = 5   # apple, banana, grape, mango, orange
DATA_DIR    = "fruits/"

datagen = ImageDataGenerator(
    rescale=1./255,
    rotation_range=20,
    width_shift_range=0.1,
    horizontal_flip=True,
    zoom_range=0.15,
    validation_split=0.2
)

train_gen = datagen.flow_from_directory(
    DATA_DIR,
    target_size=(IMG_SIZE, IMG_SIZE),
    batch_size=BATCH_SIZE,
    subset="training"
)
val_gen = datagen.flow_from_directory(
    DATA_DIR,
    target_size=(IMG_SIZE, IMG_SIZE),
    batch_size=BATCH_SIZE,
    subset="validation"
)
Python 
from tensorflow.keras.applications import MobileNetV2
from tensorflow.keras import layers, models

base = MobileNetV2(
    input_shape=(IMG_SIZE, IMG_SIZE, 3),
    include_top=False,     # drop ImageNet classifier
    weights="imagenet"
)
base.trainable = False    # freeze all base layers

inputs  = tf.keras.Input(shape=(IMG_SIZE, IMG_SIZE, 3))
x       = base(inputs, training=False)
x       = layers.GlobalAveragePooling2D()(x)
x       = layers.Dense(128, activation="relu")(x)
x       = layers.Dropout(0.3)(x)
outputs = layers.Dense(NUM_CLASSES, activation="softmax")(x)

model = models.Model(inputs, outputs)
model.summary()
Python 
model.compile(
    optimizer=tf.keras.optimizers.Adam(1e-3),
    loss="categorical_crossentropy",
    metrics=["accuracy"]
)

callbacks = [
    tf.keras.callbacks.EarlyStopping(
        patience=3,
        restore_best_weights=True
    ),
    tf.keras.callbacks.ReduceLROnPlateau(
        factor=0.5, patience=2
    )
]

history = model.fit(
    train_gen,
    epochs=10,
    validation_data=val_gen,
    callbacks=callbacks
)

import numpy as np
from tensorflow.keras.preprocessing import image

CLASS_NAMES = ["apple", "banana", "grape", "mango", "orange"]

def predict(img_path):
    img   = image.load_img(img_path, target_size=(IMG_SIZE, IMG_SIZE))
    x     = image.img_to_array(img) / 255.0
    x     = np.expand_dims(x, 0)
    probs = model.predict(x)[0]
    idx   = np.argmax(probs)
    print(f"{CLASS_NAMES[idx]}  {probs[idx]*100:.1f}%")

predict("test_banana.jpg")   # → banana  97.3%
model.save("fruit_classifier.keras")

Fundamentals

Key Concepts

Six ideas that make transfer learning so powerful.

🧩

Depthwise Separable Conv

MobileNet splits convolution into a depthwise spatial pass and a pointwise channel-mixing pass — 8–9× fewer operations than standard convolution at similar accuracy.

Efficiency
🗄️

Feature Vector

After Global Average Pooling, the 7×7×1024 feature maps collapse to a flat 1024-d vector — a compact fingerprint that your head learns to classify.

Architecture
📉

Two-phase Training

Phase 1 trains only the custom head at a high learning rate. Phase 2 fine-tunes the top base layers at LR×100 smaller. Skip Phase 1 and you risk destroying pretrained features.

Training
🔄

Data Augmentation

Artificially expands your training set via geometric and colour transforms. Dramatically reduces overfitting when you have fewer than ~1000 images per class.

Data
🎯

Softmax Output

Converts raw logits into a probability distribution that sums to 1. The argmax gives the predicted label; top-k gives ranked alternatives for confidence-aware pipelines.

Inference
⏱️

Early Stopping

Monitors validation loss and halts training when improvement stalls. Automatically restores the best checkpoint — removes the need for manual epoch tuning.

Regularization
