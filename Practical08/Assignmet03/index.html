Incremental Training Pipeline

✅ Baseline — 3 classes: apple, banana, mango
import tensorflow as tf
from tensorflow.keras.applications import MobileNetV2
from tensorflow.keras import layers, models

CLASS_NAMES_V1 = ['apple', 'banana', 'mango']
NUM_V1 = len(CLASS_NAMES_V1)  # 3

def build_model(num_classes):
    base = MobileNetV2(input_shape=(224,224,3), include_top=False, weights='imagenet')
    base.trainable = False
    x = base.output
    x = layers.GlobalAveragePooling2D()(x)
    x = layers.Dense(256, activation='relu')(x)
    x = layers.Dropout(0.35)(x)
    out = layers.Dense(num_classes, activation='softmax')(x)
    return models.Model(base.input, out)

# Phase 1 — train on 3 classes
model_v1 = build_model(NUM_V1)

⚡ Expand — add grape & orange (5 classes total)
CLASS_NAMES_V2 = ['apple', 'banana', 'mango', 'grape', 'orange']
NUM_V2 = len(CLASS_NAMES_V2)  # 5

# Strategy: re-use base feature extractor, rebuild head for 5 classes
model_v1_base = tf.keras.models.load_model('fruit_v1_3class.keras')

# Extract weights from base (up to GAP) and transfer
old_base_weights = model_v1_base.get_layer('mobilenetv2_1.00_224').get_weights()

model_v2 = build_model(NUM_V2)
model_v2.get_layer('mobilenetv2_1.00_224').set_weights(old_base_weights)

# Two-phase training for v2
# Phase A: train head only
model_v2.compile(optimizer=tf.keras.optimizers.Adam(1e-3),
                 loss='categorical_crossentropy', metrics=['accuracy'])
model_v2.fit(train_gen_5, epochs=12, validation_data=val_gen_5,
             callbacks=[tf.keras.callbacks.EarlyStopping(patience=3, restore_best_weights=True)])

# Phase B: fine-tune top 20 layers
model_v2.get_layer('mobilenetv2_1.00_224').trainable = True
for layer in model_v2.get_layer('mobilenetv2_1.00_224').layers[:-20]:
    layer.trainable = False

model_v2.compile(optimizer=tf.keras.optimizers.Adam(1e-5),
                 loss='categorical_crossentropy', metrics=['accuracy'])
h2 = model_v2.fit(train_gen_5, epochs=10, validation_data=val_gen_5,
                  callbacks=[tf.keras.callbacks.EarlyStopping(patience=3, restore_best_weights=True)])

acc_v2 = max(h2.history['val_accuracy'])
print(f"Phase 2 val accuracy: {acc_v2*100:.2f}%")
model_v2.save('fruit_v2_5class.keras')
🔬 Expand again — add strawberry & peach (7 classes total)
CLASS_NAMES_V3 = ['apple', 'banana', 'mango', 'grape', 'orange', 'strawberry', 'peach']
NUM_V3 = len(CLASS_NAMES_V3)  # 7

# Load v2 base weights
model_v2_loaded = tf.keras.models.load_model('fruit_v2_5class.keras')
v2_base_weights  = model_v2_loaded.get_layer('mobilenetv2_1.00_224').get_weights()

model_v3 = build_model(NUM_V3)
model_v3.get_layer('mobilenetv2_1.00_224').set_weights(v2_base_weights)

# Phase A: head-only training
model_v3.compile(optimizer=tf.keras.optimizers.Adam(1e-3),
                 loss='categorical_crossentropy', metrics=['accuracy'])
model_v3.fit(train_gen_7, epochs=14, validation_data=val_gen_7,
             callbacks=[tf.keras.callbacks.EarlyStopping(patience=4, restore_best_weights=True)])

# Phase B: fine-tune top 30 layers
model_v3.get_layer('mobilenetv2_1.00_224').trainable = True
for layer in model_v3.get_layer('mobilenetv2_1.00_224').layers[:-30]:

model_v3.compile(optimizer=tf.keras.optimizers.Adam(5e-6),

📊 Compare all phases side by side
import matplotlib.pyplot as plt
import numpy as np
from sklearn.metrics import classification_report

# Load all models
models_map = {
    '3-Class': ('fruit_v1_3class.keras', val_gen_3, CLASS_NAMES_V1),
    '5-Class': ('fruit_v2_5class.keras', val_gen_5, CLASS_NAMES_V2),
    '7-Class': ('fruit_v3_7class.keras', val_gen_7, CLASS_NAMES_V3),
}

results = {}
for label, (path, val_gen, cnames) in models_map.items():
    m = tf.keras.models.load_model(path)
    y_true, y_pred = [], []
    for imgs, lbls in val_gen:
        preds = m.predict(imgs, verbose=0)
        y_pred.extend(np.argmax(preds, axis=1))
        y_true.extend(np.argmax(lbls.numpy(), axis=1))
    report = classification_report(y_true, y_pred, target_names=cnames, output_dict=True)
    results[label] = report

# Plot accuracy trajectory
phases   = list(results.keys())
acc_vals = [results[p]['accuracy'] * 100 for p in phases]
n_classes = [3, 5, 7]

fig, axes = plt.subplots(1, 2, figsize=(14, 5))
fig.patch.set_facecolor('#080614')

# Accuracy vs classes
axes[0].plot(n_classes, acc_vals, 'o-', color='#7c3aed', lw=2.5, ms=10, mfc='#a78bfa')
axes[0].set_xlabel('Number of Classes', color='#a78bfa')
axes[0].set_ylabel('Val Accuracy (%)', color='#a78bfa')
axes[0].set_title('Accuracy vs. Class Count', color='white')
axes[0].set_facecolor('#0d0922')
axes[0].tick_params(colors='#666')

Findings

Key Observations

📉

Accuracy Drops with More Classes

Each new category increases confusion probability. 3→5 classes drops ~2.5%; 5→7 drops another ~2.3%. This is expected and manageable with sufficient data.

🔄

Weight Reuse Prevents Catastrophic Forgetting

Reusing the MobileNet base across phases preserves learned visual features. Starting fresh each phase would require 3× more data and training time to reach equivalent accuracy.

🎯

Similar Classes Confuse More

Peach and mango (both orange-toned) and grape and blueberry cause the most confusion. Solution: collect more varied training images for visually similar classes.

⚙️

Fine-tune Depth Should Increase

Phase 1 fine-tunes 20 layers; Phase 3 fine-tunes 30. As class diversity grows, the model needs deeper feature adaptation. Monitor validation loss closely to avoid overfitting.
